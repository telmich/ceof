SYMBOLS = ('k', 'M', 'G', 'T', 'P', 'E', 'Z', 'Y')
PREFIX = {}

for i, s in enumerate(SYMBOLS):
    PREFIX[s] = 1 << (i+1)*10

def convert_bytes(n):
    for s in reversed(SYMBOLS):
        if n >= PREFIX[s]:
            value = float(n) / PREFIX[s] 
            return '%.1f %s' % (value, s)
